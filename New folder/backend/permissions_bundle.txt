class TaskPermissions(NamedTuple):
    # العرض الأساسي
    can_view: bool

    # تعديل المهمة
    can_edit_task: bool
    can_delete_task:bool
    require_edit_basic_permission = bool
    can_transfer_task: bool
    can_change_status: bool

    # المشاركة

    can_share_remove: bool

    can_respond_urgency: bool   # الرد على الطلب (قبول/رفض)

    # التعليقات
    can_add_comment: bool
    can_edit_any_comment: bool   # تعديل أي تعليق (للأدمن ومدير البرنامج)
    can_edit_own_comment: bool   # تعديل تعليقه فقط
    can_delete_any_comment: bool
    can_delete_own_comment: bool

    # المرفقات
    can_add_attachment: bool
    can_view_attachment: bool
    can_delete_any_attachment: bool   # حذف أي مرفق (مدير البرنامج)

    can_edit_priority:bool

    def can_edit_comment(self, comment_user_id: int, current_user_id: int) -> bool:
        if self.can_edit_any_comment:
            return True
        if self.can_edit_own_comment and comment_user_id == current_user_id:
            return True
        return False

    def can_delete_comment(self, comment_user_id: int, current_user_id: int) -> bool:
        if self.can_delete_any_comment:
            return True
        if self.can_delete_own_comment and comment_user_id == current_user_id:
            return True
        return False

    def can_delete_attachment(self, attachment_user_id: int, current_user_id: int) -> bool:
        if self.can_delete_any_attachment:
            return True
        if self.can_delete_own_attachment and attachment_user_id == current_user_id:
            return True
        return False

def _has_relation(task: Task, user_id: int, relation_name: str, user_id_field: str = "user_id") -> bool:
    """التحقق من وجود المستخدم في علاقة معينة (مثل assignments, shares, comments) إذا كانت محملة."""
    if not hasattr(task, relation_name):
        return False
    for item in getattr(task, relation_name):
        if getattr(item, user_id_field) == user_id:
            return True
    return False


def _is_assignee_from_relations(task: Task, user_id: int) -> bool:
    return _has_relation(task, user_id, "assignments", "user_id")


def _is_shared_with_from_relations(task: Task, user_id: int) -> bool:
    return _has_relation(task, user_id, "shares", "shared_with_user_id")


def _is_commenter_from_relations(task: Task, user_id: int) -> bool:
    """إذا كانت task.comments محملة، نتحقق مما إذا كان المستخدم قد علق."""
    return _has_relation(task, user_id, "comments", "user_id")


def _is_department_manager(user: User, task: Task) -> bool:
    logger.debug(f"Checking department manager for user: {user.id} and task department: {task.department_id}")
    logger.debug(user)
    if not task.department_id or task.department_id != user.department_id:
        return False
    if hasattr(user, 'is_department_manager') and user.is_department_manager:
        return True
    if hasattr(user, 'job_level') and user.job_level:
        job_name = getattr(user.job_level, 'name', '')
        if job_name and job_name.lower() in ('manager', 'مدير', 'مدير قسم'):
            return True
    return False

async def get_task_permissions(

        - task.assignments (مع user إن أمكن)
        - task.shares
        - task.comments (إذا أردنا التحقق من المعلقين)
        - task.attachments (إذا أردنا التحقق من منشئ المرفق)
        - task.creator
        - task.department
    
    يمكنك استخدام task_query_with_relations() الموسعة لتشمل كل هذه العلاقات.

    if "job_level" not in user.__dict__:
        await db.refresh(user, ["job_level"])

    if AccessService._is_global_admin(user):

            can_view=True,
            can_edit_task=True,
            can_delete_task=True,
            can_share_add=True,
            can_transfer_task=True,
            can_change_status=True,
            can_share_remove=True,
            can_request_urgency=True,
            can_respond_urgency=True,
            can_convert_task=True,

            can_delete_step=True,
            can_edit_step=True,
            can_assign_executor=True,
            can_unassign_executor=True,
            can_edit_executor=True,
            can_add_comment=True,
            can_edit_any_comment=True,
            can_edit_own_comment=True,
            can_delete_any_comment=True,
            can_delete_own_comment=True,
            can_add_attachment=True,
            can_view_attachment=True,
            can_delete_any_attachment=True,
            can_delete_own_attachment=True,
            can_edit_basic=True,
            can_edit_dates=True,
            can_edit_priority=True
        )

    # تحديد الأدوار الأساسية
    is_creator = (task.created_by == user.id)
    is_assignee = _is_assignee_from_relations(task, user.id)
    is_shared = _is_shared_with_from_relations(task, user.id)
    is_commenter = _is_commenter_from_relations(task, user.id)
    is_dept_manager = _is_department_manager(user, task)
    is_prog_manager = AccessService._is_program_manager(user)
    is_global_admin = AccessService._is_global_admin(user)

    # أي شخص له علاقة بالمهمة (للعرض، إضافة تعليق، إضافة مرفق، عرض المرفق)
    has_any_relation = is_creator or is_assignee or is_shared or is_commenter or is_dept_manager or is_prog_manager

    # صلاحيات العرض
    can_view = has_any_relation

    #تعديل بيانات المهمة
    can_edit_basic=is_prog_manager or is_global_admin or is_creator or is_dept_manager
    can_edit_dates=is_prog_manager or is_global_admin or is_creator or is_dept_manager
    can_edit_priority=is_prog_manager or is_global_admin or is_creator or is_dept_manager

    # تعديل المهمة الأساسي
    can_edit_task = is_prog_manager or is_global_admin or is_creator or is_dept_manager
    can_delete_task= is_prog_manager or is_global_admin
    can_transfer_task = is_prog_manager or is_global_admin or is_creator or is_dept_manager
    can_change_status = is_prog_manager or is_global_admin or is_creator or is_dept_manager
    


    # المشاركة
    can_share_add = is_creator or is_global_admin or is_dept_manager or is_prog_manager
    can_share_remove = is_prog_manager or is_global_admin or is_dept_manager

    # طلب الاستعجال (من متطلباتك السابقة: المنشئ, مدير القسم, مدير البرنامج)
    can_request_urgency = is_creator or is_dept_manager or is_prog_manager

    # الرد على طلب الاستعجال (وفق متطلباتك: مدير البرنامج، أدمن، منشئ، مدير قسم، المنفذين، المشاركين)
    can_respond_urgency = is_prog_manager or is_global_admin or is_creator or is_dept_manager or is_assignee or is_shared

    # تحويل المهمة (حسب النص: مدير البرنامج، أدمن، منشئ، مدير قسم)
    can_convert_task = is_prog_manager or is_global_admin or is_creator or is_dept_manager

    # الخطوات
    can_add_step = is_prog_manager or is_dept_manager or is_creator or is_assignee   # منفذ يضيف خطوة
    can_delete_step = is_prog_manager or is_dept_manager or is_creator   # منشئ أو مدير القسم أو مدير البرنامج
    can_edit_step = can_delete_step   # نفس صلاحية الحذف

    # المنفذون (Assignments)
    can_assign_executor = is_prog_manager or is_dept_manager or is_creator   # + "نفس المستوى الإداري" يمكن إضافته لاحقاً
    can_unassign_executor = is_prog_manager or is_dept_manager or is_creator

    can_view_attachment = has_any_relation
    can_delete_any_attachment = is_prog_manager   # مدير البرنامج فقط (حسب طلبك: "مدير البرنامج واللى أضاف الملف")
    can_delete_own_attachment = True   # منشئ الملف يحذف ملفه

    return TaskPermissions(
        can_view=can_view,
        can_edit_task=can_edit_task,
        can_delete_task=can_delete_task,
        can_transfer_task=can_transfer_task,
        can_change_status=can_change_status,
        can_share_add=can_share_add,
        can_share_remove=can_share_remove,
        can_request_urgency=can_request_urgency,
        can_respond_urgency=can_respond_urgency,
        can_convert_task=can_convert_task,
        can_add_step=can_add_step,
        can_delete_step=can_delete_step,
        can_edit_step=can_edit_step,
        can_assign_executor=can_assign_executor,
        can_unassign_executor=can_unassign_executor,
        can_edit_executor=can_edit_executor,
        can_add_comment=can_add_comment,
        can_edit_any_comment=can_edit_any_comment,
        can_edit_own_comment=can_edit_own_comment,

class AccessService:
    def __init__(self, user: User, db: AsyncSession):
        self.user = user
        self.db = db

    async def get_user_scope_ids(self) -> List[int]:
        """
        هذه الدالة هي 'القلب' الجديد للنظام.
        تستخدم الـ Recursive CTE لجلب كل الأقسام التابعة للموظف
        بناءً على القسم الذي يقع فيه حالياً.
        """
        # منطق الـ Recursive CTE يوضع هنا
        return await DepartmentService.get_all_sub_department_ids(self.user.department_id, self.db)
    
    async def get_user_scope_ids(self) -> List[int]:
        """يجلب كل الـ IDs التابعة للمستخدم بناءً على قسمه الحالي"""
        if not self.user.department_id:
            return [self.user.department_id] # أو قائمة فارغة حسب سياستك
            
        return await DepartmentService.get_all_sub_department_ids(
            self.user.department_id, 
            self.db
        )
    
    @staticmethod
    def _is_program_manager(user: User) -> bool:
        """تحقق داخلي: هل المستخدم مدير برنامج؟"""
        return user.global_role == GlobalRole.PROGRAM_MANAGER

    @staticmethod
    def _is_global_admin(user: User) -> bool:
        """تحقق داخلي: هل المستخدم مدير نظام؟"""
        return user.global_role == GlobalRole.GLOBAL_ADMIN

    @staticmethod
    def require_pm_or_admin(user: User):
        """تحقق مركزي وصارم: يرفع استثناء إذا لم يكن المستخدم PM أو Admin"""
        if not (AccessService._is_global_admin(user) or AccessService._is_program_manager(user)):
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN, 
                detail="غير مصرح - يلزم صلاحية مدير النظام أو مدير البرنامج"
            )

    @staticmethod
    def admin_or_pm_dependency(current_user: User = Depends(get_current_user)) -> User:
        """
        Dependency Injection للتحقق من الصلاحية.
        تستخدم في الـ Routers مباشرة.
        """
        AccessService.require_pm_or_admin(current_user)
        return current_user
    
    @staticmethod
    def verify_update_permission(current_user: User, target_user_id: int):
        """تحقق مركزي: هل المستخدم أدمن/مدير برنامج؟ أو هل هو صاحب الحساب؟"""
        # نستخدم الدوال الداخلية التي عرفناها سابقاً لضمان تناسق الكود
        is_admin_or_pm = AccessService._is_global_admin(current_user) or AccessService._is_program_manager(current_user)
        is_owner = current_user.id == target_user_id
        
        if not is_admin_or_pm and not is_owner:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN, 
                detail="غير مصرح - لا تملك صلاحية تعديل بيانات هذا المستخدم"
            )
    @staticmethod
    def is_pm_or_admin(user: User) -> bool:
        """ترجع True إذا كان المستخدم مديراً أو أدمن، وإلا False."""
        return AccessService._is_global_admin(user) or AccessService._is_program_manager(user)

    @staticmethod
    def can_update_user(current_user: User, target_user_id: int) -> bool:
        """ترجع True إذا كان للمستخدم صلاحية تعديل الحساب، وإلا False."""
        is_admin = AccessService.is_pm_or_admin(current_user)
        is_owner = current_user.id == target_user_id
        return is_admin or is_owner

    @staticmethod
    def can_manage_task(user: User, task: Task) -> bool:
        """مثال إضافي: هل يمكن للمستخدم إدارة هذه المهمة؟"""
        # يمكن للمدير أو المنشئ دائماً الإدارة
        if AccessService.is_pm_or_admin(user) or task.created_by == user.id:
            return True
        return False

async def can_view_task(db: AsyncSession, user: User, task: Task) -> bool:
    """Check if user can view a task."""
    if user.global_role in (GlobalRole.GLOBAL_ADMIN,):
        return True
    # Creator
    if task.created_by == user.id:
        return True
    # Same department
    if task.department_id == user.department_id:
        return True
    # Assignee
    result = await db.execute(
        select(TaskAssignment).where(
            TaskAssignment.task_id == task.id,
            TaskAssignment.user_id == user.id
        )
    )
    if result.scalar_one_or_none():
        return True
    # Shared
    result = await db.execute(
        select(TaskShare).where(
            TaskShare.task_id == task.id,
            TaskShare.shared_with_user_id == user.id,
        )
    )
    share = result.scalar_one_or_none()
    if share and (share.expires_at is None or share.expires_at > date.today()):
        return True

    return False


async def get_task_share_permission(
    db: AsyncSession, user: User, task: Task
) -> Optional[SharePermission]:
    """Get the share permission for an external user."""
    result = await db.execute(
        select(TaskShare).where(
            TaskShare.task_id == task.id,
            TaskShare.shared_with_user_id == user.id,
        )
    )
    share = result.scalar_one_or_none()
    if share:
        return share.permission
    return None


async def can_edit_task(db: AsyncSession, user: User, task: Task) -> bool:
    """Can user edit task fields (title, desc, dates, eisenhower)?"""
    if user.global_role == GlobalRole.GLOBAL_ADMIN:
        return True
    if task.created_by == user.id:
        return True
    # Same department manager
    if task.department_id == user.department_id:
        # Any dept member can edit their tasks
        return True
    # Check share permission
    perm = await get_task_share_permission(db, user, task)
    if perm in (SharePermission.EDIT, SharePermission.MANAGE, SharePermission.ADMIN):
        return True
    # Check delegation
    if await has_delegated_permission(db, user.id, task.department_id, "modify"):
        return True
    return False


async def can_manage_task(db: AsyncSession, user: User, task: Task) -> bool:
    """Can user manage task (assign, change status, transfer)?"""
    if user.global_role == GlobalRole.GLOBAL_ADMIN:
        return True
    if task.created_by == user.id:

        return True

async def can_transfer_task(
    db: AsyncSession,
    sender: User,
    task: Task,
    target_user: User,
) -> tuple[bool, str]:
    """
    Validate same-level transfer rule.
    Returns (allowed, reason).
    """
    if sender.global_role == GlobalRole.GLOBAL_ADMIN:
        return True, ""

    # Sender must be in task's department or creator
    if task.department_id != sender.department_id and task.created_by != sender.id:
        return False, "ليس لديك صلاحية تحويل هذه المهمة"

    # Same job level check (core rule)
    if sender.job_level_id is None or target_user.job_level_id is None:
        return False, "يجب تحديد المستوى الوظيفي للمرسل والمستقبل"

    if sender.job_level_id != target_user.job_level_id:
        sender_level = sender.job_level_id
        target_level = target_user.job_level_id
        return False, (
            f"لا يمكن التحويل: المستوى الوظيفي مختلف "
            f"(المرسل: {sender_level}، المستقبل: {target_level}). "
            f"يجب التحويل لنفس المستوى الإداري."
        )

    # Can't transfer to same department (use internal assignment instead)
    if target_user.department_id == task.department_id:

async def has_delegated_permission(
    db: AsyncSession,
    user_id: int,
    department_id: int,
    permission: str,
) -> bool:
    """Check if user has a delegated permission from someone in target dept."""
    today = date.today()
    result = await db.execute(
        select(Delegation).where(
            and_(
                Delegation.delegate_id == user_id,
                Delegation.is_active == True,
                Delegation.start_date <= today,
            )
        )
    )
    delegations = result.scalars().all()
    for d in delegations:
        if d.end_date and d.end_date < today:
            continue
        if permission in d.permissions:
            return True
    return False

from fastapi import HTTPException,status
from app.models import User
from app.repositories.auth_repository import AuthRepository
from app.core.security import get_password_hash,verify_password, create_access_token, create_refresh_token,decode_token

from google.auth.transport import requests as google_requests
from app.core.config import Settings
class AuthService:
    @staticmethod
    async def authenticate_user(db, identifier, password):
        #print(get_password_hash(password))
        user = await AuthRepository.get_by_identifier(db,identifier)
        if not user or not verify_password(password, user.password_hash):
            raise HTTPException(status_code=401, detail="بيانات الدخول غير صحيحة")
        if not user.is_active:
            raise HTTPException(status_code=400, detail="الحساب غير مفعّل")
            
        await AuthRepository.update_last_login(db,user.id)
        return user

    async def google_link_logic(db, user: User, google_id: str, email: str, avatar: str):
        # 🛡️ تحقق من التكرار (Unique Constraints)
        # يمكنك إضافة دالة في الـ Repo للتحقق من عدم وجود مستخدم آخر بنفس الإيميل أو Google ID
        # ... logic here ...
        user.google_id = google_id
        user.email = email
        user.avatar_url = avatar
        await db.commit()
        return user
    
    async def authenticate_google_user(db, id_token_str: str):
        # 1. التحقق من التوكن (يمكنك نقل هذا لـ Helper في core/security)
        try:
            info = id_token.verify_oauth2_token(id_token_str, google_requests.Request(), Settings.GOOGLE_CLIENT_ID)
        except ValueError as e:
            raise HTTPException(status_code=401, detail=f"Invalid Google Token: {str(e)}")

        google_id, email = info.get("sub"), info.get("email")
        
        # 2. محاولة البحث
        user = await AuthRepository.get_by_google_id(db,google_id) or await AuthRepository.get_by_email(db,email)
        
        if not user:
            raise HTTPException(status_code=404, detail="الحساب غير مسجل بالنظام")
        
        if not user.is_active:
            raise HTTPException(status_code=400, detail="الحساب معطل")

        # 3. تحديث البيانات (Auto-link)
        return await AuthRepository.update_user_google_data(db,user, google_id, email, info.get("picture"))


    async def refresh_tokens(db, refresh_token: str):
        # 1. فك التشفير والتحقق من النوع
        payload = decode_token(refresh_token)
        if payload.get("type") != "refresh":
            raise HTTPException(status_code=400, detail="Invalid refresh token")

        # 2. جلب المستخدم
        user_id = int(payload["sub"])
        user = await AuthRepository.get_by_id(db,user_id)

        # 3. التحقق من وجود ونشاط المستخدم
        if not user or not user.is_active:
            raise HTTPException(status_code=401, detail="User not found or inactive")

        # 4. توليد التوكنز الجديدة
        return {
            "access_token": create_access_token({"sub": str(user.id)}),
            "refresh_token": create_refresh_token({"sub": str(user.id)}),
            "user": user # نرجع كائن المستخدم لتعبئة الـ Response
        }

    async def link_google_account_service(db, current_user: User, id_token_str: str):
        # 1. التحقق من التوكن (استخراج البيانات)
        id_info = id_token.verify_oauth2_token(id_token_str, google_requests.Request(), Settings.GOOGLE_CLIENT_ID)
        g_id, g_email, g_avatar = id_info.get("sub"), id_info.get("email"), id_info.get("picture")

        if not g_id or not g_email:
            raise HTTPException(status_code=400, detail="فشل استخراج بيانات Google")

        # 2. فحص الأمان (التأكد من عدم تداخل الحسابات)
        existing_by_gid = await AuthRepository.get_by_google_id(db,g_id)
        if existing_by_gid and existing_by_gid.id != current_user.id:
            raise HTTPException(status_code=400, detail="حساب Google مرتبط بموظف آخر!")

        existing_by_email = await AuthRepository.get_by_email(db,g_email)
        if existing_by_email and existing_by_email.id != current_user.id:
            raise HTTPException(status_code=400, detail="البريد الإلكتروني مستخدم من قبل موظف آخر!")

        # 3. التحديث
        current_user.google_id = g_id
        current_user.email = g_email
        current_user.avatar_url = g_avatar
        await db.commit()
        await db.refresh(current_user)
        return current_user

    async def unlink_google_account_service(self, current_user: User):
        # 1. التحقق من وجود الربط
        if not current_user.google_id:
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST, 
                detail="الحساب غير مرتبط بـ Google أصلاً لتتم عملية إلغاء الربط."
            )

        # 2. تنفيذ الإلغاء
        current_user.google_id = None
        # current_user.avatar_url = None # اختياري كما ذكرت
        
        return await AuthRepository.update_user(db,current_user)

from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update, func, or_
from sqlalchemy.orm import joinedload, load_only
from app.models import User
from app.models.DepartmentManager import DepartmentManager
from datetime import datetime

class AuthRepository:
    @staticmethod
    def basic_get_Query():
        stmt=select(User).options(
                joinedload(User.managed_departments) # جلب بيانات الإدارة التي يديرها
            )
        return stmt
    @staticmethod
    async def get_by_identifier(db: AsyncSession, identifier: str):
        stmt = AuthRepository.basic_get_Query().where(
                (User.email == identifier) | 
                (func.trim(User.employee_number) == identifier.strip())
            )
        
        result = await db.execute(stmt)
        return result.unique().scalar_one_or_none()
    
    @staticmethod
    async def get_by_google_id(db: AsyncSession, google_id: str):
        stmt=AuthRepository.basic_get_Query().options(joinedload(User.managed_departments)).where(User.google_id == google_id)
        result = await db.execute(stmt)
        return result.scalar_one_or_none()
    
    @staticmethod
    async def get_by_id(db: AsyncSession, user_id: int):
        stmt = AuthRepository.basic_get_Query().where(User.id == user_id)
        result = await db.execute(stmt)
        return result.unique().scalar_one_or_none()

    @staticmethod
    async def get_by_email(db: AsyncSession, email: str):
        stmt = AuthRepository.basic_get_Query().where(User.email == email)
        result = await db.execute(stmt)
        return result.scalar_one_or_none()

    @staticmethod
    async def update_last_login(db: AsyncSession, user_id: int):
        await db.execute(update(User).where(User.id == user_id).values(last_login=datetime.utcnow()))
        #await self.db.commit()

    @staticmethod
    async def update_user_google_data(db: AsyncSession, user: User, google_id: str, email: str, avatar: str):
        user.google_id = google_id
        user.email = email
        user.last_login = datetime.now(timezone.utc).replace(tzinfo=None)
        if avatar and not user.avatar_url:
            user.avatar_url = avatar
        #await db.commit()
        await db.refresh(user)
        return user
    
    @staticmethod
    async def update_user(db: AsyncSession, user: User):
        #await db.commit()
        await db.refresh(user)
        return user

    @staticmethod
    async def soft_delete(db: AsyncSession, location_id: int):
        loc = await AuthRepository.get_by_id(db,location_id)
        if loc:
            loc.is_active = False
            #await db.commit()
            await db.refresh(loc)
        return loc

    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
